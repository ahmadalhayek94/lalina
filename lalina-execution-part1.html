Claude Code Prompt — Copy & Paste Exactly

"I have a WordPress WooCommerce site (lalinabags.com) where product images fail to load on the shop page. They render as empty SVG placeholders due to a lazy-loading JavaScript conflict. Please create: (1) A JavaScript file that forces all placeholder images to load correctly by reading data-src attributes, (2) A functions.php snippet that disables conflicting lazy-load on shop and product pages only, (3) A CSS loading skeleton as a branded fallback while images load. Output: three complete files ready to deploy. Site uses WordPress 6.x, WooCommerce, with wp_enqueue_scripts."

lalina-image-fix.js JAVASCRIPT — Expected Output 
// Force lazy images to render on WooCommerce shop & product pages
document.addEventListener('DOMContentLoaded', function() {

  const forceLoad = () => {
    const imgs = document.querySelectorAll(
      'img[data-src], img[loading="lazy"], .woocommerce-product-gallery img'
    );
    imgs.forEach(img => {
      // Read data-src and assign to src if not a real image URL
      if (img.dataset.src && !img.src.includes('wp-content')) {
        img.src = img.dataset.src;
      }
      img.removeAttribute('loading');
      img.setAttribute('loading', 'eager');
      img.classList.remove('lazy', 'lazyload', 'not-loaded');
    });
  };

  forceLoad(); // Run immediately on DOM ready

  // Re-run when WooCommerce AJAX adds new products (filters, pagination)
  new MutationObserver(forceLoad)
    .observe(document.body, { childList: true, subtree: true });
});
functions.php — Add to child theme PHP 
// Enqueue image fix + disable WP core lazy load on commerce pages
function lalina_image_fixes() {
  wp_enqueue_script(
    'lalina-img-fix',
    get_stylesheet_directory_uri() . '/js/lalina-image-fix.js',
    [], '1.0.0', true  // true = load in footer
  );
  if ( is_shop() || is_product() || is_product_category() ) {
    add_filter( 'wp_lazy_loading_enabled', '__return_false' );
  }
}
add_action( 'wp_enqueue_scripts', 'lalina_image_fixes' );

sticky-atc.js JAVASCRIPT 
class LalinaSticky {
  constructor() {
    this.bar     = document.querySelector('.lalina-sticky-atc');
    this.trigger = document.querySelector('.single_add_to_cart_button');
    if (!this.bar || !this.trigger) return;
    this.init();
  }
  init() {
    // Show bar when main ATC button scrolls out of view
    new IntersectionObserver(
      ([e]) => this.bar.classList.toggle('visible', !e.isIntersecting),
      { rootMargin: '-60px' }
    ).observe(this.trigger);

    // Proxy click to main ATC (handles variable product validation)
    this.bar.querySelector('.sticky-atc-btn')
      .addEventListener('click', () => this.trigger.click());
  }
}
document.addEventListener('DOMContentLoaded', () => new LalinaSticky());

Task F — Exit Intent Popup
exit-intent.js JAVASCRIPT 
// Claude Code Prompt for full popup:
// "Build a complete exit-intent popup for Lalina Bags. No frameworks.
//  Dark #1a1714 background. Gold #b8975a CTA button, parchment #f7f4ef text.
//  Email capture field + '10% OFF your first order' offer.
//  Klaviyo list subscribe via their API endpoint.
//  localStorage suppression: only show once per visitor.
//  Mobile fallback: trigger after 45 seconds on-page (mouseleave unreliable mobile).
//  Smooth fade-in animation. ESC key + overlay click to close.
//  Output: complete self-contained JS + CSS + PHP hook to inject HTML."

class LalinaExitIntent {
  constructor() {
    this.shown = localStorage.getItem('lalina_exit_shown');
    if (this.shown) return; // Only show once per visitor
    this.bindEvents();
  }
  bindEvents() {
    // Desktop: trigger on cursor leaving viewport top
    document.addEventListener('mouseleave', e => {
      if (e.clientY <= 0) this.show();
    });
    // Mobile fallback: 45 seconds
    setTimeout(() => this.show(), 45000);
    // ESC key close
    document.addEventListener('keydown', e => {
      if (e.key === 'Escape') this.close();
    });
  }
  show() {
    if (this.shown) return;
    document.getElementById('lalina-exit-popup')?.classList.add('active');
    localStorage.setItem('lalina_exit_shown', 1);
    this.shown = true;
  }
  close() {
    document.getElementById('lalina-exit-popup')?.classList.remove('active');
  }
}
document.addEventListener('DOMContentLoaded', () => new LalinaExitIntent());

The Offer Principle for Lalina

Never discount the brand. Only discount the moment.

A flash sale on 3 specific SKUs for 72 hours builds urgency. A permanent 20% off everything destroys it.
Every offer should feel like an event — not a store that's always on sale.
